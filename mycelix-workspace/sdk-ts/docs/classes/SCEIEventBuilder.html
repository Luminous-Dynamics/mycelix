    Class SCEIEventBuilder<T>

    Fluent event builder for type-safe event creation

    Example

    const event = SCEIEventBuilder
      .create('calibration:prediction_recorded')
      .from('calibration')
      .withPayload({
        predictionId: 'pred-123',
        confidence: 0.8,
        domain: 'science',
        source: 'agent-1'
      })
      .withCorrelationId('corr-456')
      .build();
