    send

    • send(
          to: string,
          amount: number,
          currency?: string,
          options?: SendOptions,
      ): Promise<WalletTransaction>

      Send funds to another user

      Supports both human-readable addresses (@alice) and raw agent IDs. Uses optimistic UI - balance updates immediately, confirms in background.

      Parameters

      • to: string
      • amount: number
      • currency: string = 'MYC'
      • Optionaloptions: SendOptions

      Returns Promise<WalletTransaction>

      Example

      // Send to nickname
      await wallet.send('@alice', 50);

      // Send with memo
      await wallet.send('@bob', 100, 'MYC', { memo: 'Thanks!' });

      // Send with callbacks
      await wallet.send('@alice', 50, 'MYC', {
        onOptimistic: () => showToast('Sending...'),
        onConfirmed: () => showToast('Sent!'),
        onError: (e) => showToast(`Failed: ${e.message}`),
      });
