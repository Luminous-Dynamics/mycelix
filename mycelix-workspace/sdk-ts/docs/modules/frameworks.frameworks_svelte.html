    Namespace frameworks/svelte

    Svelte Framework Integration for Mycelix SDK

    Provides Svelte-compatible stores and utilities for building Mycelix-powered Svelte/SvelteKit applications.

    Example: Quick Start

    

{#if $mycelix}
  
    Connected as: {$identity}
    
  
    Balance: {$walletBalance.toString()} MYC
    

  
    Active Proposals
    
  {#each $proposals as proposal}
    
    
      
    {proposal.title}
    
      
    
    
  {/each}
{:else}
  
    Connecting...
    
{/if}

    Example: SvelteKit Integration

    // src/hooks.client.ts
    import { initMycelix } from '@mycelix/sdk/svelte';

    export async function init() {
      await initMycelix({
        conductorUrl: import.meta.env.VITE_CONDUCTOR_URL,
        appId: import.meta.env.VITE_APP_ID
      });
    }
